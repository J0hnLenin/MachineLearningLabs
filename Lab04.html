def minimax(data, a=0, b=1):
    new_data = copy.deepcopy(data)
    n = new_data.shape[0]
    m = new_data.shape[1]
    for i in range(m):
        minn = np.min(new_data[:, i])
        maxx = np.max(new_data[:, i])
        new_data[:, i] = (new_data[:, i] - np.full(n, minn))/(maxx - minn)*(b-a) + np.full(n, a)
    return new_data
